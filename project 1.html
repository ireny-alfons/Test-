def scatter(x,y, data=movies):
    """
    This is a function that plots scatter plots and visualize the relation between two variables
    Parameters :
    1. x: x-axis 
    2. y: y-axis 
    3. data: data frame
    """
    plt.title(f"Relation between {x} and {y}")
    plt.scatter(x=x, y=y, data=data)
    plt.xlabel(x)
    plt.ylabel(y)
    plt.show()

def movies_by_genres(movies , genres):
    """
    This function returns movies that belong to a specific genre.
    Takes the Data frame (Movies) and genre as input.
    """
    return movies[movies['genres'].str.contains(genres, case=False, na=False)]
